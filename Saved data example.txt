Read Login Data from a .txt File:
Store your login credentials in a .txt file in the format:
username=yourUsername
password=yourPassword

Use Java to read this file and extract the credentials.
Java

import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.HashMap;
import java.util.Map;

public class LoginData {
    public static Map<String, String> readLoginData(String filePath) throws IOException {
        Map<String, String> loginData = new HashMap<>();
        BufferedReader reader = new BufferedReader(new FileReader(filePath));
        String line;
        while ((line = reader.readLine()) != null) {
            String[] parts = line.split("=");
            if (parts.length == 2) {
                loginData.put(parts, parts);
            }
        }
        reader.close();
        return loginData;
    }
}
